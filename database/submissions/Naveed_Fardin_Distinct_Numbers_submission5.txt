#include<iostream>
#include<vector>
#include<set>
using namespace std;
int main()
{
    int n;
    cin>>n;
    vector<int> v;
    set<int> s;
    for(int i=0;i<n;i++)
    {
        int p;
        cin>>p;
        v.push_back(p);
    }
 
    for(int X:v)
    {
        s.insert(X); 
    }
 
    cout<<s.size();
}
