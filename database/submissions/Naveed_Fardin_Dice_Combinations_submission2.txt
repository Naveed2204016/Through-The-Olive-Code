#include<bits/stdc++.h>
using namespace std;
const int mod=1e9+7;
int main()
{
    int x;
    cin>>x;
    int dp[x+1]={0};
    dp[0]=1;
    for(int i=1;i<=x;i++)
    {
        for(int j=1;j<=6;j++)
        {
            if(j<=i)
            dp[i]=(dp[i]+dp[i-j])%mod;
        }
    }
    cout<<dp[x]<<endl;
}
